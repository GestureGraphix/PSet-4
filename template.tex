\section{Problem 1: Stewart 11.2.20}
\qs{}{Determine whether the series is convergent or divergent
by expressing $S_{n}$ as a telescoping sum. If it is convergent find its sum
    \begin{center}
        \[ \sum_{n=1}^{\infty} \ln\left(\frac{n}{n+1}\right)\] 
    \end{center}
}

\sol{
    \begin{center}
        \[ \sum_{n=1}^{\infty} \ln\left(\frac{n}{n+1}\right) = \sum_{n=1}^{\infty} \ln(n)-\ln(n+1)\]
        \[ S_{n} =  \left(\ln(1) - \ln(1 + 1)\right) + \left(\ln(2) - \ln(1 + 2)\right) + \ldots + \left(\ln(n) - \ln(n + 1)\right) \] 
        \[ S_{n} =  \left(\ln(1) - \ln(2)\right) + \left(\ln(2) - \ln(3)\right) + \ldots + \left(\ln(n) - \ln(n + 1)\right) \] 
        \[ S_{n} = \left(\ln(1) - \ln(n+1)\right)\] 
        \[ \lim_{n \to \infty} S_{n} = \lim_{n \to \infty} \left(\ln(1) - \ln(n+1)\right) = - \infty \]
        \[ \text{diverges}\]  
    \end{center}
}

\section{Problem 4}
\qs{}{Use the Comparison Test to decide whether the series 
    \begin{center} 
        \[ \sum_{k=3}^{\infty} \frac{2+(-1)^{k}}{k}\] 
    \end{center}
converges or diverges. Explain your reasoning. (In particular, explain why it is valid to use comparison
and what you are comparing to).}

\sol{
    \begin{center}
        \[ \frac{1}{k} \leq \frac{2 + (-1)^{k}}{k} \leq \frac{3}{k}\] 
        \[ a_{k} = \frac{1}{k} \]
        \[ b_{k} = \frac{2 + (-1)^{k}}{k} \]  
        \[ \sum_{n=3}^{\infty} \frac{1}{k} \text{ diverges}\] 
    \end{center}
    Since the series is always greater that or equal to the harmonic series, which diverges, it can be said that 
    this series also diverges using the comparison test. $a_{k} \leq b_{k}$ and it diverges so $b_k$ diverges as well. 
}

\section{Problem 5}
\qs{}{A ball is dropped out of a window from a height of 100 feet. Assume that each time the ball falls from
a height $h$ that it rebounds to 30$\text{\% }$ of height $h$. Assuming the ball bounces indefinitely, determine the
total distance it travels.}

\sol{
    \begin{center}
        \[ \sum_{n = 1}^{\infty} (200)(0.3^{n}) \]  
        \[ \sum_{n= 1}^{\infty} ar^{n} = \frac{a}{1-r}\]
        \[ a = 100\]
        \[ r = 0.3\]
        \[ \frac{100}{1 - 0.3} = \frac{1000}{7} \approx 142.857 \text{ feet} \]   
    \end{center}
}
